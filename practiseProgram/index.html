Program 2 : Find Sum Of First n Natural Numbers in O(1)

#include <stdio.h>
int main()
{
int num, sum = 0;
printf("Enter the number : ");
scanf("%d", &num);
for (int i = 1; i <= num; i++)
{
sum += i;
}
printf("The sum till %d is = %d",num,sum);
return 0;
}

Program 3 : Find the largest number in an array

#include <stdio.h>
int main()
{
int arr[] = {1, 4, 6, 3, 47, 1000, 77, 56, 345, 56, 678};
int lar = arr[0];
for (int i = 1; i < (sizeof(arr)) / (sizeof(int)); i++)
{
if (arr[i] > lar)
{
lar = arr[i];
}
}

printf("The largest element in arr is = %d", lar);
return 0;
}
